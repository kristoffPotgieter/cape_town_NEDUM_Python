    """Round to nearest decimal number."""
    return round(round(x / a) * a, 2)

        df_3 = dictio[flood_type + '_100yr']

                  df_2.flood_depth, df_2.sim_poor, df_2.sim_midpoor,
                  df_2.sim_midrich, df_2.sim_rich,
